Create `frontend/src/lib/data/properties.ts` with React cache() for deduplicated data fetching:

1. Import `cache` from 'react', `getPayloadInstance` from '@/lib/payload'
2. Export `getPublishedProperties` - cached function that fetches published properties with depth=2 (featuredImage, gallery)
3. Export `getFeaturedProperties` - cached function that fetches up to 6 featured/published properties
4. Export `getPropertiesBySlug` - cached function that fetches a single property by slug (where slug is generated from title via Payload)
5. Use `depth: 2` for relationship fields
6. Return properly typed PaginatedResponse from getPayload find() results

The file structure:
```typescript
import { cache } from 'react';
import { getPayloadInstance } from '@/lib/payload';
import type { PaginatedResponse } from '@/types';
import type { Property } from '@real-estate-saas/shared';

export const getPublishedProperties = cache(async (options?: {
  limit?: number;
  offset?: number;
}): Promise<PaginatedResponse<Property>> => {
  const payload = await getPayloadInstance();
  return payload.find({
    collection: 'properties',
    where: { status: { equals: 'published' } },
    limit: options?.limit ?? 20,
    offset: options?.offset ?? 0,
    depth: 2,
  });
});

export const getFeaturedProperties = cache(async (): Promise<Property[]> => {
  const payload = await getPayloadInstance();
  const result = await payload.find({
    collection: 'properties',
    where: { status: { equals: 'published' } },
    limit: 6,
    depth: 2,
  });
  return result.docs as unknown as Property[];
});

export const getPropertyBySlug = cache(async (slug: string): Promise<Property | null> => {
  const payload = await getPayloadInstance();
  const result = await payload.find({
    collection: 'properties',
    where: { slug: { equals: slug }, status: { equals: 'published' } },
    limit: 1,
    depth: 3,
  });
  return (result.docs[0] as unknown as Property) ?? null;
});
```

Note: Payload generates slug automatically from title (via Slug field type or auto-generation). Use the slug field in where clause.
